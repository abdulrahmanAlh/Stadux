## Install

```bash
npm install --save state-any-where
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'state-any-where'
import 'state-any-where/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
